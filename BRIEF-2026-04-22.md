    `French DNB exam iconographic document. ${testContent.imageDescription}. ` +

testContent.imageUrl = `https://image.pollinations.ai/prompt/${pollinationsPrompt}?width=800&height=600&nologo=true&seed=${Date.now()}`;

- Remplacement par hint méthodologique : "💡 Méthode : Observe attentivement ce document..."
- Spinner de chargement + fallback si image non disponible

### 6. Spinner + fallback image
```tsx
<img src={testContent.imageUrl}
  onLoad={e => { (e.target as HTMLElement).parentElement?.querySelector('.img-loading')?.remove(); }}
  onError={e => {
    const el = e.target as HTMLImageElement;
    el.style.display = 'none';
    const fb = el.parentElement?.querySelector('.img-fallback') as HTMLElement;
    if (fb) fb.style.display = 'flex';
  }}
/>
<div className="img-loading absolute inset-0 ...">
  <div className="w-8 h-8 border-4 border-blue-200 border-t-blue-500 rounded-full animate-spin" />
  <p>Chargement du document...</p>
</div>
<div className="img-fallback absolute inset-0 hidden ...">
  <span>🖼️</span><p>Document non disponible</p>
</div>
```
